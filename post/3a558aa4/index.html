import numpy as np
import tensorflow as tf

def leak_relu(x, alpha=0.1):
    return tf.maximum(alpha * x, x)


class Yolo(object):
    def __init__(self, weights_file):
        self.verbose = False
        # detection params
        self.S = 7  # cell size
        self.B = 2  # boxes_per_cell
        self.classes = ["aeroplane", "bicycle", "bird", "boat", "bottle",
                        "bus", "car", "cat", "chair", "cow", "diningtable",
                        "dog", "horse", "motorbike", "person", "pottedplant",
                        "sheep", "sofa", "train", "tvmonitor"]
        self.C = len(self.classes)  # number of classes
        # grid cell相对于全图左上角(0,0)的偏移量 取值从[0-6]
        # 全图左上角cell(0,0) 右下角cell(6,6)
        self.x_offset = np.transpose(np.reshape(np.array([np.arange(self.S)] * self.S * self.B),
                                                [self.B, self.S, self.S]), [1, 2, 0])
        self.y_offset = np.transpose(self.x_offset, [1, 0, 2])

        self.threshold = 0.2  # class-score小于0.2的，重置为0
        self.iou_threshold = 0.5 # 针对于每一个class, 98个box进行nms时，iou小于0.5的score (98*20)被设置为0

    def detect_from_file(self, image_file, imshow=True, deteted_boxes_file="boxes.txt",
                         detected_image_file="detected_image.jpg"):
        """Do detection given a image file"""
        # read image
        image = cv2.imread(image_file)
        img_h, img_w, _ = image.shape
        print("height = ", img_h)  # 424
        print("width = ", img_w)  # 640
        # predicts = self._detect_from_image(image)
        predicts = np.load("person.npy")  # (1470,)
        print("predicts.shape", predicts.shape)
        predict_boxes = self._interpret_predicts(predicts, img_h, img_w)
        self.show_results(image, predict_boxes, imshow, deteted_boxes_file, detected_image_file)

    def _detect_from_image(self, image):
        """Do detection given a cv image"""
        img_resized = cv2.resize(image, (448, 448))
        img_RGB = cv2.cvtColor(img_resized, cv2.COLOR_BGR2RGB)
        img_resized_np = np.asarray(img_RGB)
        _images = np.zeros((1, 448, 448, 3), dtype=np.float32)
        _images[0] = (img_resized_np / 255.0) * 2.0 - 1.0  # (-1.0,  1.0)
        predicts = self.sess.run(self.predicts, feed_dict={self.images: _images})[0]
        return predicts

    def _interpret_predicts(self, predicts, img_h, img_w):
        """Interpret the predicts and get the detetction boxes"""

        # predicts = z = x*w + b, no activation   取值有正有负
        idx1 = self.S * self.S * self.C
        idx2 = idx1 + self.S * self.S * self.B
        print("idx1 = ", idx1)
        print("idx2 = ", idx2)

        # class prediction
        class_probs = np.reshape(predicts[:idx1], [self.S, self.S, self.C])
        # confidence
        confs = np.reshape(predicts[idx1:idx2], [self.S, self.S, self.B])
        # boxes -> (x, y, w, h)
        boxes = np.reshape(predicts[idx2:], [self.S, self.S, self.B, 4])

        print("\nclass_probs.shape = ", class_probs.shape)  # (7, 7, 20)
        print("confs.shape = ", confs.shape)  # (7, 7, 2)
        print("boxes.shape = ", boxes.shape)  # (7, 7, 2, 4)

        print("\n  class_probs")
        # print(class_probs[0,0,:])  # class_probs有正有负

        print("\n  confs")
        # print(confs[:,:,0])  # confidence的预测值有负有正 -0.5,  0.1, 0.9

        print("\n cell x,y; image w,h")
        print(boxes[0, 0, 0, :])
        print(boxes[0, 0, 1, :])
        print(boxes[6, 6, 0, :])
        print(boxes[6, 6, 1, :])

        # convert the x, y to the coordinates relative to the top left point of the image
        # x,y是相对于cell左上角的坐标，取值在[0-1]之间
        # x,y的基础上增加cell相对于整个图像的左上角的x_offset,y_offset[0-6]，得到相对于全图的左上角坐标,取值在[0-7]之间
        # x,y除以cell的大小7,得到相对于全图左上角的坐标,取值归一化到[0-1]区间
        boxes[:, :, :, 0] += self.x_offset
        boxes[:, :, :, 1] += self.y_offset
        boxes[:, :, :, :2] /= self.S

        print("\n image x,y; image w,h")
        print(boxes[0, 0, 0, :])
        print(boxes[0, 0, 1, :])
        print(boxes[6, 6, 0, :])
        print(boxes[6, 6, 1, :])

        # 预测是实际上是sqrt_w,sqrt_h，取值在[0-1]区间，所以需要平方得到真正的w,h
        # w,h取值是相对于全图的大小比例，取值在[0-1]区间。
        # eg. sqrt_w, sqrt_h = 0.8 ===>实际w,h = 0.8*0.8 = 0.64
        # the predictions of w, h are the square root
        boxes[:, :, :, 2:] = np.square(boxes[:, :, :, 2:])

        print("\n square w,h")
        print(boxes[0, 0, 0, :])
        print(boxes[0, 0, 1, :])
        print(boxes[6, 6, 0, :])
        print(boxes[6, 6, 1, :])

        # x,y,w,h现在都是相对于全图的比例，取值在[0-1]之间。乘以图像的大小得到，实际的x,y,w,h
        # multiply the width and height of image
        boxes[:, :, :, 0] *= img_w
        boxes[:, :, :, 1] *= img_h
        boxes[:, :, :, 2] *= img_w
        boxes[:, :, :, 3] *= img_h

        print("\n real x,y,w,h")
        print(boxes[0, 0, 0, :])
        print(boxes[0, 0, 1, :])
        print(boxes[6, 6, 0, :])
        print(boxes[6, 6, 1, :])

        print("==============================================")
        print("\n 23 grid, conficence(b0) (b1)")
        print(confs[3, 2, :])  # 23 grid,

        print("\n 23 grid, class_probs")
        print(class_probs[3, 2, :])  # 23 grid, class_probs

        print("\n 23 grid, conficence(b0) * class_probs = score[46]  HIT")
        print(confs[3, 2, 0] * class_probs[3, 2, :])

        print("\n 23 grid, conficence(b1) * class_probs = score[47]")
        print(confs[3, 2, 1] * class_probs[3, 2, :])
        print("==============================================")

        # 7*7*2  ===> 7*7*2*1
        e_confs = np.expand_dims(confs, -1)  # last dim
        print("e_confs.shape=", e_confs.shape)

        # 7*7*20 ===>7*7*1*20
        e_class_probs = np.expand_dims(class_probs, 2)
        print("e_class_probs.shape=", e_class_probs.shape)

        # scores: class-specific confidence score
        scores = np.expand_dims(confs, -1) * np.expand_dims(class_probs, 2)
        print("scores.shape=", scores.shape)  # (7, 7, 2, 20)

        print("\n  scores")
        # print(scores[0,0,0:,:]) # score = class_prob* confidence 有正有负

        # 98个box的class-score, box坐标
        scores = np.reshape(scores, [-1, self.C])  # 7*7*2*20====>98*20
        boxes = np.reshape(boxes, [-1, 4])  # 7*7*2*4 ===> 98*4

        print("\n scores[46]")  # 46 box ===> 23 grid  score[3,2,]
        print(scores[46, :])
        print("argmax = ", np.argmax(scores[46, :]), " max = ", np.max(scores[46, :]))

        print("\n scores[72]")
        print(scores[72, :])
        print("argmax = ", np.argmax(scores[72, :]), " max = ", np.max(scores[72, :]))

        print("\n boxes")
        print("\n boxes[46]")
        print(boxes[46, :])

        print("\n boxes[72]")
        print(boxes[72, :])

        # class-score小于0.2的，重置为0
        # filter the boxes when score < threhold
        scores[scores < self.threshold] = 0.0

        # non max suppression
        self._non_max_suppression(scores, boxes)

        # report the boxes
        predict_boxes = []  # (class_name, x, y, w, h, class_score)

        print("\n scores 0")
        max_idxs = np.argmax(scores, axis=1)  # 98*20 ===> 98 max score index (class-index)
        print("\n max_ids")
        print(max_idxs)
        for i in range(len(scores)):  # 98-b
            max_idx = max_idxs[i]  # 0, 14, 11
            max_score = scores[i, max_idx]
            if max_score > 0.0:  # score[0,0] == 0
                class_name = self.classes[max_idx]
                print("   hit box i = ", i, " class index = ", max_idx, "class name= ", class_name, " score= ",
                      max_score)
                # hit box i =  46  class index =  14 class name=  person  score=  0.6021214
                # hit box i =  72  class index =  11 class name=  dog  score=  0.4548718

                predict_boxes.append((class_name, boxes[i, 0], boxes[i, 1], boxes[i, 2], boxes[i, 3], max_score))
                # class_name, x,y,w,h, class_score
        return predict_boxes

    def _non_max_suppression(self, scores, boxes):
        """Non max suppression"""
        # scores = 98*20
        # boxes = 98*4
        # for each class
        for c in range(self.C):
            sorted_idxs = np.argsort(scores[:, c])
            if c == 14 or c == 11:
                print("\n =====before nms======")
                print(scores[:, c])
                print(sorted_idxs)
            last = len(sorted_idxs) - 1  # 97,96,....3,2,1,0
            while last > 0:
                if scores[sorted_idxs[last], c] < 1e-6:
                    break
                for i in range(last):
                    if scores[sorted_idxs[i], c] < 1e-6:
                        continue
                    if self._iou(boxes[sorted_idxs[i]], boxes[sorted_idxs[last]]) > self.iou_threshold:
                        scores[sorted_idxs[i], c] = 0.0  # 对于每一个class的98个box,如果iou大于0.5,则标记score为0
                last -= 1
            if c == 14 or c == 11:
                print("\n =====after nms======")
                print(scores[:, c])

    def _iou(self, box1, box2):
        """Compute the iou of two boxes"""
        # box (cx,cy,w,h)
        inter_w = np.minimum(box1[0] + 0.5 * box1[2], box2[0] + 0.5 * box2[2]) - \
                  np.maximum(box1[0] - 0.5 * box2[2], box2[0] - 0.5 * box2[2])
        inter_h = np.minimum(box1[1] + 0.5 * box1[3], box2[1] + 0.5 * box2[3]) - \
                  np.maximum(box1[1] - 0.5 * box2[3], box2[1] - 0.5 * box2[3])
        if inter_h < 0 or inter_w < 0:
            inter = 0
        else:
            inter = inter_w * inter_h
        union = box1[2] * box1[3] + box2[2] * box2[3] - inter
        return inter / union

    def show_results(self, image, results, imshow=False, deteted_boxes_file=None,
                     detected_image_file=None):
        """Show the detection boxes"""
        img_cp = image.copy()  # origin image (no resize)
        # draw boxes
        for i in range(len(results)):
            x = int(results[i][1])
            y = int(results[i][2])
            w = int(results[i][3])
            h = int(results[i][4])
            print("cx,cy,w,h =", x, y, w, h)
            # 141/2 = 70.5 float;   141/2 = 70 int;
            (x1, y1) = (x - w // 2, y - h // 2)
            (x2, y2) = (x + w // 2, y + h // 2)

            if self.verbose:
                print("   class: %s, [cx, cy, w, h]=[%d, %d, %d, %d], confidence=%f" % (results[i][0],
                                                                                        x, y, w, h, results[i][-1]))
            cv2.rectangle(img_cp, (x1, y1), (x2, y2), (0, 255, 0), 2)
            cv2.putText(img_cp, results[i][0] + ' : %.2f' % results[i][5], (x1 + 5, y1 - 7),
                        cv2.FONT_HERSHEY_SIMPLEX, 0.5, (0, 0, 0), 1)
        if detected_image_file:
            cv2.imwrite(detected_image_file, img_cp)

def draw_cell_index(image, w, h):
    S = 7
    B = 2
    # h, w = 424,640
    x_offset = np.transpose(np.reshape(np.array([np.arange(7)] * 7), [7, 7]), [0, 1])
    y_offset = np.transpose(x_offset, [1, 0])
    # print(x_offset)
    # print(y_offset)
    x_offset = (x_offset / S * w).astype(np.int32)
    y_offset = (y_offset / S * h).astype(np.int32)
    # print(x_offset) # 7*7
    # print(y_offset) # 7*7
    cell_w = w / 7
    cell_h = h / 7

    for row in range(S):
        for col in range(7):
            cell_index = row * S + col
            x1 = x_offset[row][col]
            y1 = y_offset[row][col]
            x2 = int(x1 + cell_w)
            y2 = int(y1 + cell_h)
            cx = int((x1 + x2) / 2)
            cy = int((y1 + y2) / 2)

            cv2.rectangle(image, (x1, y1), (x2, y2), (0, 0, 0), 2)
            text = str(row) + "," + str(col)
            text = str(cell_index)
            cv2.putText(image, text, (x1, y2 - 5), cv2.FONT_HERSHEY_COMPLEX, 0.8, (255, 0, 0), 2)

    return image


def draw_cell_box_index(image, w, h):
    S = 7
    B = 2
    # h, w = 424,640
    x_offset = np.transpose(np.reshape(np.array([np.arange(7)] * 7), [7, 7]), [0, 1])
    y_offset = np.transpose(x_offset, [1, 0])
    # print(x_offset)
    # print(y_offset)
    x_offset = (x_offset / S * w).astype(np.int32)
    y_offset = (y_offset / S * h).astype(np.int32)
    # print(x_offset) # 7*7
    # print(y_offset) # 7*7
    cell_w = w / 7
    cell_h = h / 7

    for row in range(S):
        for col in range(S):
            cell_index = row * 7 + col
            x1 = x_offset[row][col]
            y1 = y_offset[row][col]
            x2 = int(x1 + cell_w)
            y2 = int(y1 + cell_h)
            cx = int((x1 + x2) / 2)
            cy = int((y1 + y2) / 2)
            # cv2.rectangle(image, (x1,y1), (x2,y2), (0, 255, 0), 2)

            for b in range(B):
                box_index = cell_index * B + b
                text = str(box_index)
                if b == 0:
                    cy_up = int(cy - cell_h / 4)
                    cv2.putText(image, text, (cx, cy_up), cv2.FONT_HERSHEY_COMPLEX, 0.5, (255, 255, 0), 1)
                else:
                    cy_down = int(cy + cell_h / 4)
                    cv2.putText(image, text, (cx, cy_down), cv2.FONT_HERSHEY_COMPLEX, 0.5, (255, 255, 0), 1)

    return image


def draw_box(image, box):
    x, y, w, h = box

    (x1, y1) = (x - w // 2, y - h // 2)
    (x2, y2) = (x + w // 2, y + h // 2)

    cv2.rectangle(image, (x1, y1), (x2, y2), (0, 255, 0), 2)
    cv2.circle(image, (x, y), 3, (0, 255, 0), 5)
    return image
